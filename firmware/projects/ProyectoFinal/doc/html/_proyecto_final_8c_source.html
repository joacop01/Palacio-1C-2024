256static void ProcessSignal(void *pvParameters)
257{
258 while (true)
259 {
260 ulTaskNotifyTake(pdTRUE, portMAX_DELAY);
261
262 if(on)
263 {
264 readNextSample(&hr_monitor);
265 processLatestSample(&hr_monitor);
266 if (count == 0)
267 {
268 UartSendString(UART_PC, "S\r\n");
269 }
270
271 if(sawStartOfBeat(&hr_monitor))
272 {
273 if(periodo_refractario <= 0 )
274 {
275 uint8_t *msg = UartItoa((uint32_t)getInterBeatIntervalMs(&hr_monitor), 10);
276 UartSendString(UART_PC, (char*)msg);
277 UartSendString(UART_PC, "\r\n");
278 periodo_refractario = PERIODO_REF;
279 }
280 }
281 periodo_refractario--;
282 count++;
283
284 if (count == MUESTRAS_1_MIN)
285 {
286 UartSendString(UART_PC, "E\r\n");
287 count = 0;
288 }
289 }
290 }
291}
