162static void SuministroAgua(void *pvParameters)

179static void ControlpH(void *pvParameters)
180{
181 while(true){
182
183 ulTaskNotifyTake(pdTRUE, portMAX_DELAY);
184
185 AnalogInputReadSingle(CH1, &pH); //conversión analógica digital
186 pH = (pH * 14) / 3000; //conversión del valor de tensión (mV) a pH
187
188 if(pH < 6) //si el pH es menor a 6
189 {
190 GPIOOn(GPIO_BOMBA_SOL_BASICA); //prendemos la bomba de solucion básica
191 }
192 else // si el pH es mayor o igual a 6
193 {
194 GPIOOff(GPIO_BOMBA_SOL_BASICA); //apagamos la bomba de solucion básica
195 }
196
197 if(pH > 6.7) //si el pH es mayor a 6.7
198 {
199 GPIOOn(GPIO_BOMBA_SOL_ACIDA); //prendemos la bomba de solución ácida
200 }
201 else // si el pH es menor o igual a 6.7
202 {
203 GPIOOff(GPIO_BOMBA_SOL_ACIDA); //apagamos la bomba de solución ácida
204 }
205 }
206}
